    public static void printList(Node start){

    public static Node convertToCircularList(int[] arr){
        Node temp = null;
        Node start = temp;
        for(int i=0;i<arr.length;i++){
            int val = arr[i];
            if(i==0){
                temp = new Node(val);
                temp.next = start;
                start = temp;
            }else{
                temp.next  = new Node(val);
                temp = temp.next;
                temp.next = start;
            }

        }
        return start;
